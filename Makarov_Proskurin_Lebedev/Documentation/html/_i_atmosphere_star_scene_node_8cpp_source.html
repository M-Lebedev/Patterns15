51 void IAtmosphereStarSceneNode::generateMesh()
52 {
53  BufferSun->Vertices.clear();
54  BufferSun->Indices.clear();
55 
56  BufferMoon->Vertices.clear();
57  BufferMoon->Indices.clear();
58 
59  BufferSun->Vertices.reallocate( 4 );
60  BufferSun->Indices.reallocate( 6 );
61 
62  BufferMoon->Vertices.reallocate( 4 );
63  BufferMoon->Indices.reallocate( 6 );
64 
65  BufferSun->Indices.push_back(0);
66  BufferSun->Indices.push_back(2);
67  BufferSun->Indices.push_back(1);
68  BufferSun->Indices.push_back(0);
69  BufferSun->Indices.push_back(3);
70  BufferSun->Indices.push_back(2);
71 
72  BufferMoon->Indices.push_back(0);
73  BufferMoon->Indices.push_back(2);
74  BufferMoon->Indices.push_back(1);
75  BufferMoon->Indices.push_back(0);
76  BufferMoon->Indices.push_back(3);
77  BufferMoon->Indices.push_back(2);
78 
79  video::S3DVertex vtx = video::S3DVertex(0.f, 0.f, 0.f,
80  0.f, 0.f, 0.f,
81  video::SColor(255,255,255,255), 0.f, 0.f);
82 
83  for (u16 i = 0; i < 4; i++)
84  {
85  BufferSun->Vertices.push_back(vtx);
86  BufferMoon->Vertices.push_back(vtx);
87  }
88 
89  BufferSun->Vertices[0].TCoords.set(1.0f, 1.0f);
90  BufferSun->Vertices[1].TCoords.set(1.0f, 0.0f);
91  BufferSun->Vertices[2].TCoords.set(0.0f, 0.0f);
92  BufferSun->Vertices[3].TCoords.set(0.0f, 1.0f);
93 
94  BufferMoon->Vertices[0].TCoords.set(1.0f, 1.0f);
95  BufferMoon->Vertices[1].TCoords.set(1.0f, 0.0f);
96  BufferMoon->Vertices[2].TCoords.set(0.0f, 0.0f);
97  BufferMoon->Vertices[3].TCoords.set(0.0f, 1.0f);
98 
99  BufferSun->setHardwareMappingHint(scene::EHM_STREAM, scene::EBT_VERTEX);
100  BufferMoon->setHardwareMappingHint(scene::EHM_STREAM, scene::EBT_VERTEX);
101 }

114 void IAtmosphereStarSceneNode::render()
115 {
116  video::IVideoDriver* driver = SceneManager->getVideoDriver();
117  scene::ICameraSceneNode* camera = SceneManager->getActiveCamera();
118 
119  if (!camera || !driver)
120  return;
121 
122  // make billboard look to camera
123  core::vector3df campos = camera->getAbsolutePosition();
124  core::vector3df target = camera->getTarget();
125  core::vector3df up = camera->getUpVector();
126  core::vector3df view = target - campos;
127  view.normalize();
128 
129  core::vector3df horizontal = up.crossProduct(view);
130  horizontal.normalize();
131 
132  core::vector3df vertical = horizontal.crossProduct(view);
133  vertical.normalize();
134 
135  core::dimension2d<f32> oldSize = Size;
136 
137  horizontal *= (oldSize.Width / 16);
138  vertical *= (oldSize.Height / 16);
139 
140  // Draw the sun
141 
142  BufferSun->Vertices[0].Pos.set(SunPosition + horizontal + vertical);
143  BufferSun->Vertices[1].Pos.set(SunPosition + horizontal - vertical);
144  BufferSun->Vertices[2].Pos.set(SunPosition - horizontal - vertical);
145  BufferSun->Vertices[3].Pos.set(SunPosition - horizontal + vertical);
146 
147  view *= -1.0f;
148 
149  for (s32 i=0; i<4; ++i)
150  BufferSun->Vertices[i].Normal.set(view);
151 
152  // Draw the moon
153 
154  BufferMoon->Vertices[0].Pos.set(MoonPosition + horizontal + vertical);
155  BufferMoon->Vertices[1].Pos.set(MoonPosition + horizontal - vertical);
156  BufferMoon->Vertices[2].Pos.set(MoonPosition - horizontal - vertical);
157  BufferMoon->Vertices[3].Pos.set(MoonPosition - horizontal + vertical);
158 
159  view *= -1.0f;
160 
161  for (s32 i=0; i<4; ++i)
162  BufferMoon->Vertices[i].Normal.set(view);
163 
164  if ( !camera->isOrthogonal() )
165  {
166  core::matrix4 mat(AbsoluteTransformation);
167  mat.setTranslation(camera->getAbsolutePosition());
168  driver->setTransform(video::ETS_WORLD, mat);
169 
170  driver->setMaterial(BufferSun->Material);
171  driver->drawMeshBuffer(BufferSun);
172  driver->setMaterial(BufferMoon->Material);
173  driver->drawMeshBuffer(BufferMoon);
174  }
175 
176  // for debug purposes only:
177  if ( DebugDataVisible )
178  {
179  video::SMaterial m;
180  m.Lighting = false;
181  driver->setMaterial(m);
182 
183  if ( DebugDataVisible & scene::EDS_NORMALS )
184  {
185  // draw normals
186  const f32 debugNormalLength = SceneManager->getParameters()->getAttributeAsFloat(scene::DEBUG_NORMAL_LENGTH);
187  const video::SColor debugNormalColor = SceneManager->getParameters()->getAttributeAsColor(scene::DEBUG_NORMAL_COLOR);
188  driver->drawMeshBufferNormals(BufferSun, debugNormalLength, debugNormalColor);
189  driver->drawMeshBufferNormals(BufferMoon, debugNormalLength, debugNormalColor);
190  }
191 
192  // show mesh
193  if ( DebugDataVisible & scene::EDS_MESH_WIRE_OVERLAY )
194  {
195  m.Wireframe = true;
196  driver->setMaterial(m);
197 
198  driver->drawMeshBuffer(BufferSun);
199  driver->drawMeshBuffer(BufferMoon);
200  }
201  }
202 }
