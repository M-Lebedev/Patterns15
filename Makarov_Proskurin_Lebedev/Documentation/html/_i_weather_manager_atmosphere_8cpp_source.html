82 void IWeatherManagerAtmosphere::OnAnimate(u32 timeMs)
83 {
84  if (FirstUpdate)
85  {
86  LastAnimationTime = timeMs;
87 
88  FirstUpdate = false;
89  }
90 
91  // get time
92  f32 timeDiff = static_cast<f32>( timeMs - LastAnimationTime );
93 // std::cout << timeDiff << std::endl;
94  LastAnimationTime = timeMs;
95 
96  Theta += (180.f * 100.f) / (86400.f * DaySpeed) * timeDiff;
97 }

100 void IWeatherManagerAtmosphere::render()
101 {
102  modulo360();
103  f32 sinOfPhi = sinf(Phi * core::DEGTORAD);
104  f32 cosOfPhi = cosf(Phi * core::DEGTORAD);
105 
106  f32 sinOfTheta = sinf(Theta * core::DEGTORAD);
107  f32 cosOfTheta = cosf(Theta * core::DEGTORAD);
108 
109  f32 x = Radius * cosOfTheta * cosOfPhi;
110  f32 y = Radius * sinOfTheta;
111  f32 z = Radius * cosOfTheta * sinOfPhi;
112 
113  core::vector3df sun_place = core::vector3df(x, y, z);
114  if ( Rotation != 0.f )
115  sun_place.rotateYZBy(Rotation, smgr->getRootSceneNode()->getAbsolutePosition());
116 
117  core::vector3df billboardPosition;
118  billboardPosition = sun_place;
119  Bill->setSunPosition(billboardPosition);
120 
121  billboardPosition = -sun_place;
122  Bill->setMoonPosition(billboardPosition);
123 
124 // Sunlight->setPosition(sun_place * 10.f + core::vector3df(256.f*10.f/2.f, 0.f, 256.f*10.f/2.f));
125 
126  SkyPixel = SkyImage->getPixel(static_cast<u32>(Theta), 0); // Y = 123
127 
128  smgr->setAmbientLight(video::SColorf(SkyPixel.getRed(), SkyPixel.getGreen(), SkyPixel.getBlue(), 1.f));
129 
131 // smgr->setShadowColor(video::SColor(255, SkyPixel.getRed(), SkyPixel.getGreen(), SkyPixel.getBlue()));
132 // Sunlight->getLightData().DiffuseColor = video::SColorf(video::SColor(SkyPixel.getRed(), SkyPixel.getGreen(), SkyPixel.getBlue(), 1.f));
133 
134  Sky->setUvX(Theta / 360.f);
135 }
