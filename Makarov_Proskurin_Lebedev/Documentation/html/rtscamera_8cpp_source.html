53 bool RTSCamera::OnEvent(const SEvent& event)
54 {
55  if (!InputReceiverEnabled)
56  return false;
57 
58  dimension2d<u32> ssize = SceneManager->getVideoDriver()->getScreenSize();
59  if(event.EventType == EET_MOUSE_INPUT_EVENT)

85  case EMIE_RMOUSE_LEFT_UP:
86  MouseKeys[2] = false;

88  case EMIE_MMOUSE_LEFT_UP:
89  MouseKeys[1] = false;

93  IVideoDriver* driver = SceneManager->getVideoDriver();
94  if (driver)

353 void RTSCamera::animate()
354 {
355  f32 nRotX = rotX;
356  f32 nRotY = rotY;
357  f32 nZoom = currentZoom;
358 
359  vector3df translate(oldTarget);
360  vector3df tvectX = Pos - Target;
361  tvectX = tvectX.crossProduct(UpVector);
362  tvectX.normalize();
363 
364  //Zoom
365  if (isMouseButtonDown(MOUSE_BUTTON_RIGHT) && isMouseButtonDown(MOUSE_BUTTON_LEFT))
366  {
367  if (!zooming)
368  {
369  zoomStartX = MousePos.X;
370  zoomStartY = MousePos.Y;
371  zooming = true;
372  nZoom = currentZoom;
373  }
374  else
375  {
376  f32 old = nZoom;
377  nZoom += (zoomStartX - MousePos.X) * zoomSpeed * 100;
378 
379 // f32 targetMinDistance = 0.1f;
380 // if (nZoom < targetMinDistance)
381 // nZoom = targetMinDistance;
382 
383  if (nZoom < targetMinDistance)
384  nZoom = targetMinDistance;
385  else if (nZoom > targetMaxDistance)
386  nZoom = targetMaxDistance;
387 
388 
389  if (nZoom < 0)
390  nZoom = old;
391  }
392  }
393  else
394  {
395  if (zooming)
396  {
397  f32 old = currentZoom;
398  currentZoom = currentZoom + (zoomStartX - MousePos.X ) * zoomSpeed;
399  nZoom = currentZoom;
400 
401  if (nZoom < 0)
402  nZoom = currentZoom = old;
403  }
404  zooming = false;
405  }
406 
407  //Rotate
408  if(isMouseButtonDown(MOUSE_BUTTON_LEFT) && !zooming)
409  {
410  if (!rotating)
411  {
412  rotateStartX = MousePos.X;
413  rotateStartY = MousePos.Y;
414  rotating = true;
415  nRotX = rotX;
416  nRotY = rotY;
417  }
418  else
419  {
420  nRotX += (rotateStartX - MousePos.X) * rotateSpeed;
421  nRotY += (rotateStartY - MousePos.Y) * rotateSpeed;
422  }
423  }
424  else
425  {
426  if (rotating)
427  {
428  rotX = rotX + (rotateStartX - MousePos.X) * rotateSpeed;
429  rotY = rotY + (rotateStartY - MousePos.Y) * rotateSpeed;
430  nRotX = rotX;
431  nRotY = rotY;
432  }
433 
434  rotating = false;
435  }
436 
437  //Translate
438  if(isMouseButtonDown(MOUSE_BUTTON_RIGHT) && !zooming)
439  {
440  if (!translating)
441  {
442  translateStartX = MousePos.X;
443  translateStartY = MousePos.Y;
444  translating = true;
445  }
446  else
447  {
448  translate += tvectX * (translateStartX - MousePos.X) * translateSpeed;
449  translate.X += tvectX.Z * (translateStartY - MousePos.Y) * translateSpeed;
450  translate.Z -= tvectX.X * (translateStartY - MousePos.Y) * translateSpeed;
451 
452  oldTarget = translate;
453  }
454  }
455  else if (isKeyDown(KEY_KEY_W) || isKeyDown(KEY_UP) && !zooming)
456  {
457  if (!translating)
458  translating = true;
459  else
460  {
461  vector3df movevector = getPosition() - getTarget();
462  movevector.Y = 0;
463  movevector.normalize();
464 
465  setPosition(getPosition() - movevector * translateSpeed);
466  setTarget(getTarget() - movevector * translateSpeed);
467  updateAbsolutePosition();
468  }
469  }
470  else if (isKeyDown(KEY_KEY_S) || isKeyDown(KEY_DOWN) && !zooming)
471  {
472  if (!translating)
473  translating = true;
474  else
475  {
476  vector3df movevector = getPosition() - getTarget();
477  movevector.Y = 0;
478  movevector.normalize();
479 
480  setPosition(getPosition() + movevector * translateSpeed);
481  setTarget(getTarget() + movevector * translateSpeed);
482  updateAbsolutePosition();
483  }
484  }
485  else if (isKeyDown(KEY_KEY_A) || isKeyDown(KEY_LEFT) && !zooming)
486  {
487  if (!translating)
488  translating = true;
489  else
490  {
491  vector3df totargetvector = getPosition() - getTarget();
492  totargetvector.normalize();
493  vector3df crossvector = totargetvector.crossProduct(getUpVector());
494  vector3df strafevector = crossvector.normalize();
495 
496  setPosition(getPosition() - strafevector * translateSpeed);
497  setTarget(getTarget() - strafevector * translateSpeed);
498  updateAbsolutePosition();
499  }
500  }
501  else if (isKeyDown(KEY_KEY_D) || isKeyDown(KEY_RIGHT) && !zooming)
502  {
503  if (!translating)
504  translating = true;
505  else
506  {
507  vector3df totargetvector = getPosition() - getTarget();
508  totargetvector.normalize();
509  vector3df crossvector = totargetvector.crossProduct(getUpVector());
510  vector3df strafevector = crossvector.normalize();
511 
512  setPosition(getPosition() + strafevector * translateSpeed);
513  setTarget(getTarget() + strafevector * translateSpeed);
514  updateAbsolutePosition();
515  }
516  }
517  else
518  {
519  translating = false;
520 
521  if (!translating && !zooming && !rotating)
522  {
523  //Mouse Coordinates go from 0 to 1 on both axes
524  if (MousePos.X < 0.05) //Up
525  {
526  vector3df totargetvector = getPosition() - getTarget();
527  totargetvector.normalize();
528  vector3df crossvector = totargetvector.crossProduct(getUpVector());
529  vector3df strafevector = crossvector.normalize();
530 
531  setPosition(getPosition() - strafevector * translateSpeed);
532  setTarget(getTarget() - strafevector * translateSpeed);
533  updateAbsolutePosition();
534  }
535  else if (MousePos.X > 0.95) //Down
536  {
537  vector3df totargetvector = getPosition() - getTarget();
538  totargetvector.normalize();
539  vector3df crossvector = totargetvector.crossProduct(getUpVector());
540  vector3df strafevector = crossvector.normalize();
541 
542  setPosition(getPosition() + strafevector * translateSpeed);
543  setTarget(getTarget() + strafevector * translateSpeed);
544  updateAbsolutePosition();
545  }
546  else if (MousePos.Y < 0.05) //Up
547  {
548  vector3df movevector = getPosition() - getTarget();
549  movevector.Y = 0;
550  movevector.normalize();
551 
552  setPosition(getPosition() - movevector * translateSpeed);
553  setTarget(getTarget() - movevector * translateSpeed);
554  updateAbsolutePosition();
555  }
556  else if (MousePos.Y > 0.95) //Down
557  {
558  vector3df movevector = getPosition() - getTarget();
559  movevector.Y = 0;
560  movevector.normalize();
561 
562  setPosition(getPosition() + movevector * translateSpeed);
563  setTarget(getTarget() + movevector * translateSpeed);
564  updateAbsolutePosition();
565  }
566  }
567  }
568 
569  //Set Position
570  Target = translate;
571 
572  Pos.X = nZoom + Target.X;
573  Pos.Y = Target.Y;
574  Pos.Z = Target.Z;
575 
576  Pos.rotateXYBy(nRotY, Target);
577  Pos.rotateXZBy(-nRotX, Target);
578 
579  //Correct Rotation Error
580  UpVector.set(0,1,0);
581  UpVector.rotateXYBy(-nRotY, core::vector3df(0,0,0));
582  UpVector.rotateXZBy(-nRotX+180.f, core::vector3df(0,0,0));
583 }
