48 void IAtmosphereSkySceneNode::generateMesh()
49 {
50  Buffer->Vertices.clear();
51  Buffer->Indices.clear();
52 
53  Buffer->Vertices.reallocate( Faces + 1 );
54  Buffer->Indices.reallocate( Faces * 3 );
55 
56  // This is just a cone
57 
58  f32 Angle = 0.f; //start positions
59  const f32 AngleStep = 2.f * core::PI / Faces; //angle to add
60 
61  video::S3DVertex vtx = video::S3DVertex(0.f, Radius, 0.f,
62  0.f, -1.f, 0.f,
63  video::SColor(255,255,255,255), 0.f, 0.1f); // If you want to change the sky color, you can do it here.
64 
65  Buffer->Vertices.push_back(vtx);
66 
67  for (u16 i = 1; i < Faces + 1; i++)
68  {
69  vtx.Pos.X = cosf(Angle) * Radius; //vertice x coord
70  vtx.Pos.Y = 0.f;
71  vtx.Pos.Z = sinf(Angle) * Radius; //vertice z coord
72  vtx.Normal = vtx.Pos;
73  vtx.Normal.normalize();
74  vtx.TCoords = core::vector2df(0.f, 1.f);
75  Buffer->Vertices.push_back(vtx);
76 
77  Angle += AngleStep; // we neglish the precision of a floating summ. f32 angle = angle2 * i; This would be better
78  }
79 
80  for (u16 i = 1; i < Faces; i++)
81  {
82  Buffer->Indices.push_back(0);
83  Buffer->Indices.push_back(i);
84  Buffer->Indices.push_back(i + 1);
85  }
86 
87  Buffer->Indices.push_back(0);
88  Buffer->Indices.push_back(Faces);
89  Buffer->Indices.push_back(1);
90 
91  Buffer->setHardwareMappingHint(scene::EHM_STATIC, scene::EBT_VERTEX);
92 }

105 void IAtmosphereSkySceneNode::render()
106 {
107  video::IVideoDriver* driver = SceneManager->getVideoDriver();
108  scene::ICameraSceneNode* camera = SceneManager->getActiveCamera();
109 
110  if (!camera || !driver)
111  return;
112 
113  //update uv maping
114  for(s32 i = 1; i < Faces + 1; i++)
115  {
116  Buffer->Vertices[i].TCoords.set(UvX, 1.f);
117  }
118 
119  Buffer->Vertices[0].TCoords.set(UvX, 0.1f); // (UvX, 0.f); does a black point in the sky.
120 
121  if ( !camera->isOrthogonal() )
122  {
123  core::matrix4 mat(AbsoluteTransformation);
124  mat.setTranslation(camera->getAbsolutePosition());
125 
126  driver->setTransform(video::ETS_WORLD, mat);
127 
128  driver->setMaterial(Buffer->Material);
129  driver->drawMeshBuffer(Buffer);
130  }
131 
132  // for debug purposes only:
133  if ( DebugDataVisible )
134  {
135  video::SMaterial m;
136  m.Lighting = false;
137  driver->setMaterial(m);
138 
139  if ( DebugDataVisible & scene::EDS_NORMALS )
140  {
141  // draw normals
142  const f32 debugNormalLength = SceneManager->getParameters()->getAttributeAsFloat(scene::DEBUG_NORMAL_LENGTH);
143  const video::SColor debugNormalColor = SceneManager->getParameters()->getAttributeAsColor(scene::DEBUG_NORMAL_COLOR);
144  driver->drawMeshBufferNormals(Buffer, debugNormalLength, debugNormalColor);
145  }
146 
147  // show mesh
148  if ( DebugDataVisible & scene::EDS_MESH_WIRE_OVERLAY )
149  {
150  m.Wireframe = true;
151  driver->setMaterial(m);
152 
153  driver->drawMeshBuffer(Buffer);
154  }
155  }
156 }
